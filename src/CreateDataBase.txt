DROP TABLE IF EXISTS Authors
DROP TABLE IF EXISTS Publications
DROP TABLE IF EXISTS Venues
CREATE TABLE Authors (Name VARCHAR(100) PRIMARY KEY NOT NULL, NumOfPubl INTEGER);
CREATE TABLE Venues (VenID VARCHAR(100) NOT NULL PRIMARY KEY, Name VARCHAR(100) NOT NULL);
CREATE TABLE Publications (Title VARCHAR(100) NOT NULL PRIMARY KEY, NumOfAuth INTEGER, YearOfOcc YEAR, AuthorID VARCHAR(100) NOT NULL, VenID VARCHAR(100) NOT NULL, FOREIGN KEY (AuthorID) REFERENCES Authors (Name), FOREIGN KEY (VenID) REFERENCES Venues (VenID));
