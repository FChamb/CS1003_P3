DROP TABLE IF EXISTS Authors
DROP TABLE IF EXISTS Publications
DROP TABLE IF EXISTS Venues
CREATE TABLE Authors (Name VARCHAR(100) PRIMARY KEY NOT NULL, NumOfPubl INTEGER, AuthorID VARCHAR(100) NOT NULL);
CREATE TABLE Venues (PublVenID VARCHAR(100) NOT NULL PRIMARY KEY, Name VARCHAR(100) NOT NULL);
CREATE TABLE Publications (Title VARCHAR(100) NOT NULL PRIMARY KEY, NumOfAuth INTEGER, YearOfOcc YEAR, AuthorID VARCHAR(100) NOT NULL, PublVenID VARCHAR(100) NOT NULL, FOREIGN KEY (AuthorID) REFERENCES Authors (AuthorID), FOREIGN KEY (PublVenID) REFERENCES Venues (PublVenID));
