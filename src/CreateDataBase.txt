DROP TABLE IF EXISTS Authors
DROP TABLE IF EXISTS Publications
DROP TABLE IF EXISTS Venues
CREATE TABLE Authors (Name VARCHAR(100) PRIMARY KEY NOT NULL, NumOfPubl INTEGER, NumOfCoAuth INTEGER, AuthorID INTEGER);
CREATE TABLE Publications (AuthorID INTEGER PRIMARY KEY, Title VARCHAR(100) NOT NULL, NumOfAuth INTEGER, PublVenID INTEGER);
CREATE TABLE Venues (PublVenID INTEGER PRIMARY KEY, Name VARCHAR(100) NOT NULL, YearOfOcc YEAR);
